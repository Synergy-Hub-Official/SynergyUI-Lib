Getting Started

1. Load the Library

local SynergyUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/Synergy-Hub-Official/Scripts/refs/heads/main/SynergyUI.lua"))()

The library is hosted on GitHub and can be loaded directly into any Roblox script (LocalScript recommended).

2. Create Your First Window

local window = SynergyUI:CreateWindow({
    Title = "My Hub",
    AccentColor = Color3.fromRGB(0, 255, 100),
    ConfigFile = "myhub_config.json",
    ToggleKey = Enum.KeyCode.RightShift,
    Parent = nil, -- default: CoreGui (or PlayerGui in Studio)
    CornerRadius = 6,
    PaddingHorizontal = 10,
    EnableCtrlTab = true
})

Tip: All theme parameters (BackgroundColor, SidebarColor, Font, etc.) are also available.

Tabs

window:CreateTab(name, iconUrl) → returns an elements table with all control constructors.

local mainTab = window:CreateTab("Main")
local settingsTab = window:CreateTab("Settings", "rbxassetid://123456789")

Each tab's elements can create labels, buttons, toggles, and more.

Controls Reference

Button

Clickable button with optional tooltip.

els:CreateButton({ Name = "Click", Callback = function() end, Tooltip = "..." })

Toggle

Switch with state saving. Tooltip supported.

els:CreateToggle({ Name = "ESP", CurrentValue = true, Flag = "ESP", Callback = function(s) end })

Slider

Number slider with optional increment and input box.

els:CreateSlider({ Name = "Speed", Range = {16,250}, Increment = 1, CurrentValue = 50, Callback = function(v) end })

Dropdown

Expandable list. Returns object with SetOptions/SetValue/GetValue.

local dd = els:CreateDropdown({ Name = "Mode", Options = {"A","B"}, Flag = "mode" })
dd:SetOptions({"X","Y"}) -- dynamic options

Checklist

Multi-select list. Returns object with SetOptions/GetSelected/SetSelected.

local cl = els:CreateChecklist({ Name = "Features", Options = {"X","Y"} })
cl:SetSelected({"X"})

Text Input

Free text field. Tooltip supported.

els:CreateTextInput({ Name = "Message", CurrentText = "", Callback = function(t) end })

Number Input

Validated numeric input.

els:CreateNumberInput({ Name = "Value", CurrentValue = 100, Callback = function(n) end })

Keybind

Click to set any key. Callback returns the key name.

els:CreateKeybind({ Name = "Ultimate", CurrentKeybind = "F", Callback = function(k) end })

Color Picker

RGB sliders with live preview.

els:CreateColorPicker({ Name = "Color", Color = Color3.new(1,0,0), Callback = function(c) end })

Radio Group

Single-select radio buttons.

els:CreateRadioGroup({ Name = "Theme", Options = {"Dark","Light"}, CurrentValue = "Dark", Callback = function(v) end })

Utilities

  • els:CreateLabel(text)
  • els:CreateSeparator()
  • els:CreateSection(name)
  • els:CreateParagraph({Title, Content, Image})
els:CreateParagraph({
    Title = "Info",
    Content = "Rich text paragraph",
    Image = "rbxassetid://12345" -- optional
})

💡 Tooltip is available on Button, Toggle, Slider, Dropdown, TextInput, Keybind, ColorPicker.

Complete Working Example

A full script demonstrating all controls, including programmatic usage, paragraph, and notification with position.

-- SynergyUI Full Example
local SynergyUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/Synergy-Hub-Official/Scripts/refs/heads/main/SynergyUI.lua"))()

local window = SynergyUI:CreateWindow({
    Title = "Synergy Test Hub",
    AccentColor = Color3.fromRGB(0, 255, 100),
    ConfigFile = "synergy_test.json",
    ToggleKey = Enum.KeyCode.RightShift,
    EnableCtrlTab = true
})

local mainTab = window:CreateTab("Main")
local movementTab = window:CreateTab("Movement")
local visualsTab = window:CreateTab("Visuals")

-- Main Tab
mainTab:CreateSection("General")
mainTab:CreateLabel("Welcome to SynergyUI")

mainTab:CreateToggle({
    Name = "Enable ESP",
    CurrentValue = true,
    Flag = "ESPEnabled",
    Callback = function(state) print("ESP:", state) end,
    Tooltip = "Draws boxes around players"
})

mainTab:CreateButton({
    Name = "Force Respawn",
    Callback = function()
        SynergyUI:Notify("Respawned!", 2, Color3.fromRGB(0, 255, 100), "BottomRight")
    end
})

mainTab:CreateSeparator()
mainTab:CreateParagraph({
    Title = "What is SynergyUI?",
    Content = "A modern UI library with full config saving, flags, and programmatic control.",
    Image = "rbxassetid://123456789" -- optional
})

-- Movement Tab
movementTab:CreateSection("Movement Options")
movementTab:CreateToggle({
    Name = "Speed Boost",
    Flag = "SpeedBoost",
    Callback = function(s) print("Boost:", s) end
})

movementTab:CreateSlider({
    Name = "WalkSpeed",
    Range = {16, 250},
    Increment = 1,
    CurrentValue = 50,
    Flag = "WalkSpeed",
    Callback = function(v) print("Speed set to", v) end
})

local modeDropdown = movementTab:CreateDropdown({
    Name = "Movement Mode",
    Options = {"Default", "Bhop", "AirWalk"},
    CurrentOption = "Default",
    Flag = "MovementMode",
    Callback = function(opt) print("Mode:", opt) end
})

-- dynamically change dropdown options later
task.wait(3)
modeDropdown:SetOptions({"Default", "Bhop", "AirWalk", "Teleport"})

-- Checklist with programmatic access
local checklist = movementTab:CreateChecklist({
    Name = "Enabled Features",
    Options = {"Infinite Jump", "No Fall Damage", "Anti-AFK"},
    CurrentSelected = {"Infinite Jump"},
    Flag = "MovementFeatures",
    Callback = function(selected) print("Selected:", selected) end
})

-- Visuals Tab
visualsTab:CreateSection("Visual Settings")
visualsTab:CreateColorPicker({
    Name = "ESP Color",
    Color = Color3.fromRGB(0, 255, 100),
    Flag = "ESPColor",
    Callback = function(c) print("Color changed") end
})

visualsTab:CreateRadioGroup({
    Name = "Theme Style",
    Options = {"Neon", "Minimal", "Dark"},
    CurrentValue = "Neon",
    Flag = "ThemeStyle",
    Callback = function(v) print("Theme:", v) end
})

visualsTab:CreateTextInput({
    Name = "Custom Message",
    CurrentText = "Hello",
    Flag = "CustomMsg",
    Callback = function(t) SynergyUI:Notify(t, 3) end
})

visualsTab:CreateKeybind({
    Name = "Kill Aura Key",
    CurrentKeybind = "F",
    Flag = "KillAuraKey",
    Callback = function(key) print("Key set to", key) end
})

-- Programmatic control example
task.wait(1)
window.Flags.ESPEnabled:SetValue(false)   -- disable ESP via flag
print("ESP enabled?", window.Flags.ESPEnabled:GetValue())

window:SaveConfig() -- manually save
SynergyUI:Notify("UI Ready", 4, Color3.fromRGB(0, 255, 100), "TopRight")

Advanced Features

Notifications with Position

SynergyUI:Notify("Message", duration, Color3, position) -- position: "TopRight", "TopLeft", "BottomRight", "BottomLeft"

Config System

Automatically saves position, accent, and all controls with a Flag when ConfigFile is provided.

window:SaveConfig("custom.json")   -- override filename
window:LoadConfig("custom.json")
window:Destroy()                    -- cleanup connections and GUI

Global Settings

You can also change the accent color dynamically after creation:

window:SetAccent(Color3.fromRGB(255, 0, 0))

Programmatic Control

Every control with a Flag is automatically registered in window.Flags[flagName] with methods GetValue() and SetValue(newValue).

-- Assume you created a toggle with Flag = "MyToggle"
local toggleFlag = window.Flags.MyToggle
print(toggleFlag:GetValue())   -- current boolean
toggleFlag:SetValue(true)       -- triggers callback and saves

-- For dropdowns, GetValue returns the selected string
local mode = window.Flags.MovementMode:GetValue()
window.Flags.MovementMode:SetValue("Bhop")

-- For sliders, returns number
local speed = window.Flags.WalkSpeed:GetValue()
window.Flags.WalkSpeed:SetValue(120)

-- Checklist returns a table of selected options
local features = window.Flags.MovementFeatures:GetValue()
window.Flags.MovementFeatures:SetValue({"Infinite Jump", "Anti-AFK"})

Additionally, Dropdown and Checklist creation return an object with extra methods:

local dd = tab:CreateDropdown({ ... })
dd:SetOptions({"New", "Options"})   -- update list
dd:SetValue("New")
local current = dd:GetValue()

local cl = tab:CreateChecklist({ ... })
cl:SetOptions({"A","B","C"})
cl:SetSelected({"A","C"})
local selected = cl:GetSelected()
