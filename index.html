Getting Started

1. Load the Library

local SynergyUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/Synergy-Hub-Official/Scripts/refs/heads/main/SynergyUI.lua"))()

The library is hosted on GitHub and can be loaded directly into any Roblox script (LocalScript recommended).

2. Create Your First Window

local window = SynergyUI:CreateWindow({
    Title = "My Hub",
    AccentColor = Color3.fromRGB(0, 255, 100),
    ConfigFile = "myhub_config.json",
    ToggleKey = Enum.KeyCode.RightShift
})

Complete Working Example

Here is a fully functional script that demonstrates every major feature of SynergyUI. Copy and run it in Studio to see it live.

-- SynergyUI Full Example Script
local SynergyUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/Synergy-Hub-Official/Scripts/refs/heads/main/SynergyUI.lua"))()

local window = SynergyUI:CreateWindow({
    Title = "Synergy Test Hub",
    AccentColor = Color3.fromRGB(0, 255, 100),
    BackgroundColor = Color3.fromRGB(15, 15, 15),
    SidebarColor = Color3.fromRGB(20, 20, 20),
    ConfigFile = "synergy_test.json",
    ToggleKey = Enum.KeyCode.RightShift,
    EnableCtrlTab = true
})

local mainTab = window:CreateTab("Main")
local movementTab = window:CreateTab("Movement")
local visualsTab = window:CreateTab("Visuals")

local els = mainTab

els:CreateSection("General")
els:CreateLabel("Welcome to SynergyUI")

els:CreateToggle({
    Name = "Enable ESP",
    CurrentValue = true,
    Flag = "ESPEnabled",
    Callback = function(state)
        print("ESP toggled:", state)
    end,
    Tooltip = "Draws boxes around players"
})

els:CreateButton({
    Name = "Force Respawn",
    Callback = function()
        SynergyUI:Notify("Player respawned", 2, Color3.fromRGB(0, 255, 100))
    end,
    Tooltip = "Instantly respawns local player"
})

els:CreateSeparator()

els = movementTab

els:CreateSection("Movement Options")

els:CreateToggle({
    Name = "Speed Boost",
    CurrentValue = false,
    Flag = "SpeedBoost",
    Callback = function(state)
        print("Speed boost:", state and "ON" or "OFF")
    end
})

els:CreateSlider({
    Name = "WalkSpeed",
    Range = {16, 250},
    Increment = 1,
    CurrentValue = 50,
    Flag = "WalkSpeed",
    Callback = function(value)
        print("WalkSpeed set to", value)
    end
})

els:CreateDropdown({
    Name = "Movement Mode",
    Options = {"Default", "Bhop", "AirWalk", "Teleport"},
    CurrentOption = "Default",
    Flag = "MovementMode",
    Callback = function(option)
        print("Movement mode changed to:", option)
    end
})

els:CreateChecklist({
    Name = "Enabled Features",
    Options = {"Infinite Jump", "No Fall Damage", "Anti-AFK"},
    CurrentSelected = {"Infinite Jump"},
    Flag = "MovementFeatures",
    Callback = function(selected)
        print("Selected features:", selected)
    end
})

els = visualsTab

els:CreateSection("Visual Settings")

els:CreateColorPicker({
    Name = "ESP Color",
    Color = Color3.fromRGB(0, 255, 100),
    Flag = "ESPColor",
    Callback = function(color)
        print("ESP color updated")
    end
})

els:CreateRadioGroup({
    Name = "Theme Style",
    Options = {"Neon", "Minimal", "Dark"},
    CurrentValue = "Neon",
    Flag = "ThemeStyle",
    Callback = function(value)
        print("Theme changed to", value)
    end
})

els:CreateTextInput({
    Name = "Custom Message",
    CurrentText = "Hello from SynergyUI",
    Flag = "CustomMsg",
    Callback = function(text)
        SynergyUI:Notify(text, 3)
    end
})

els:CreateKeybind({
    Name = "Kill Aura Key",
    CurrentKeybind = "F",
    Flag = "KillAuraKey",
    Callback = function(key)
        print("Kill aura key set to", key)
    end
})

SynergyUI:Notify("SynergyUI loaded successfully!", 4, Color3.fromRGB(0, 255, 100), "TopRight")

This script creates a fully working UI with tabs, every control type, config saving, notifications, and keyboard toggle.

local mainTab = window:CreateTab("Main")

Each tab returns an elements table containing all control creation functions.

Controls Reference

Button

Simple clickable button with optional tooltip.

els:CreateButton({ Name = "Click Me", Callback = function() end, Tooltip = "Optional" })

Toggle

On/off switch with state saving.

els:CreateToggle({ Name = "Toggle", CurrentValue = false, Flag = "MyFlag", Callback = function(s) end })

Slider

Number slider with optional input box.

els:CreateSlider({ Name = "Speed", Range = {16,250}, Increment = 1, CurrentValue = 50, Callback = function(v) end })

Dropdown

Expandable list selector.

els:CreateDropdown({ Name = "Mode", Options = {"A","B"}, CurrentOption = "A", Callback = function(opt) end })

Checklist

Multi-select checklist.

els:CreateChecklist({ Name = "Features", Options = {"X","Y"}, CurrentSelected = {"X"}, Callback = function(t) end })

Text Input

Free text field.

els:CreateTextInput({ Name = "Message", CurrentText = "Hello", Callback = function(t) end })

Number Input

Validated numeric input.

els:CreateNumberInput({ Name = "Value", CurrentValue = 100, Callback = function(n) end })

Keybind

Click to set any key.

els:CreateKeybind({ Name = "Trigger", CurrentKeybind = "F", Callback = function(k) end })

Color Picker

RGB sliders with preview.

els:CreateColorPicker({ Name = "Color", Color = Color3.new(1,0,0), Callback = function(c) end })

Radio Group

Single-select radio buttons.

els:CreateRadioGroup({ Name = "Choice", Options = {"One","Two"}, CurrentValue = "One", Callback = function(v) end })

Utilities

  • els:CreateSeparator()
  • els:CreateSection(name)

Advanced Features

Notifications

SynergyUI:Notify("Message", duration, Color3, position)

Config System

Automatically saves position, accent, and all control states when ConfigFile is provided.

window:SaveConfig("custom.json")
window:LoadConfig("custom.json")

Programmatic Control Access

Every control with a Flag can be accessed later via window.Flags[flagName]
